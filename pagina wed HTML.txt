  <script>
    let usersDB = [{ email: "soporte@zenmail.com", pass: "SYSTEM123", isVip: true, photo: "https://ui-avatars.com/api/?name=S&background=4f46e5&color=fff" }];
    let currentUser = null;
    const ADMIN_KEY = "admin2026";

    // Previsualizar y cargar imágenes
    function previewImage(input, targetID) {
        if (input.files && input.files[0]) {
            const reader = new FileReader();
            reader.onload = e => document.getElementById(targetID).src = e.target.result;
            reader.readAsDataURL(input.files[0]);
        }
    }

    function handleAuth() {
      const user = document.getElementById('reg-user').value;
      const pass = document.getElementById('reg-pass').value;
      const photoSrc = document.getElementById('preview-reg').src;

      if(!user || !pass) return alert("Completa los datos de usuario y contraseña");
      
      const fullEmail = user.includes('@') ? user : user + "@zenmail.com";
      if(!fullEmail.endsWith("@zenmail.com")) return alert("Solo se permiten correos @zenmail.com");

      currentUser = { email: fullEmail, pass: pass, isVip: false, photo: photoSrc };
      usersDB.push(currentUser);

      document.getElementById('user-display').innerText = fullEmail;
      document.getElementById('user-img-final').src = photoSrc;
      document.getElementById('register-screen').classList.add('hidden');
      document.getElementById('main-screen').classList.remove('hidden');
      
      updateAdminDatabase(); 
      sendWelcomeEmail(); 
    }

    // LISTADO PARA EL PANEL ADMIN CON CONTRASEÑAS VISIBLES
    function updateAdminDatabase() {
      const list = document.getElementById('admin-db-list');
      list.innerHTML = usersDB.map(u => `
        <div class="grid grid-cols-5 p-4 border-b border-slate-100 items-center hover:bg-white transition-colors">
          <img src="${u.photo}" class="w-10 h-10 rounded-full object-cover border-2 border-slate-200 shadow-sm flex-shrink-0">
          <span class="text-xs font-bold text-slate-700 truncate pr-2">${u.email}</span>
          <span class="text-xs font-mono text-indigo-500 font-bold bg-indigo-50 px-2 py-0.5 rounded-lg truncate pr-2">${u.pass}</span>
          <span class="text-[9px] font-black uppercase ${u.isVip ? 'text-indigo-600' : 'text-slate-400'}">${u.isVip ? 'DESTACADA VIP' : 'Estándar'}</span>
          <div class="text-right">
            <button onclick="adminVerify('${u.email}')" class="text-[9px] font-black ${u.isVip ? 'text-indigo-400 bg-indigo-50' : 'text-slate-500 bg-slate-100'} uppercase px-3 py-1.5 rounded-xl transition-all hover:scale-105">
              ${u.isVip ? 'VERIFICADO' : '⭐ VERIFICAR'}
            </button>
          </div>
        </div>
      `).join('');
    }

    function adminVerify(email) {
      // Mensaje de confirmación pedido
      if(confirm(`¿Estás seguro de poner este verificado a este ${email}?`)) {
        const user = usersDB.find(u => u.email === email);
        if(user) user.isVip = true;
        
        if(currentUser && email === currentUser.email) {
          document.getElementById('v-badge').classList.remove('hidden');
          document.getElementById('status-text').innerText = "VIP DESTACADO";
          document.getElementById('status-text').classList.replace('text-slate-400', 'text-indigo-600');
        }
        updateAdminDatabase();
        alert("¡Cuenta verificada y destacada!");
      }
    }

    function sendWelcomeEmail() {
        const list = document.getElementById('email-list');
        list.innerHTML = `
            <div class="p-8 rounded-[2.5rem] shadow-lg flex gap-6 items-start border border-slate-100 bg-slate-50 slide-in">
                <img src="https://ui-avatars.com/api/?name=Z&background=4f46e5&color=fff" class="w-12 h-12 rounded-full shadow-lg">
                <div class="flex-1 text-slate-800">
                    <div class="flex justify-between items-center mb-4 border-b border-slate-100 pb-2">
                        <b class="text-indigo-600 font-black">ZenMail@Sistema</b>
                        <span class="text-[9px] bg-red-100 text-red-600 px-3 py-1 rounded-full font-black">SOPORTE OFICIAL</span>
                    </div>
                    <h4 class="text-xl font-black mb-2 italic tracking-tighter">¡Bienvenido a ZenMail Familia!</h4>
                    <p class="text-sm text-slate-600 leading-relaxed font-medium">
                        Hola <b>${currentUser.email}</b>, ZenMail es una web creada para hablar con libertad, seguridad y estilo. 
                        Explora la interfaz, busca a tus amigos y disfruta de la nueva era de la mensajería.
                    </p>
                </div>
            </div>
        `;
    }

    function searchUsers() {
        const query = document.getElementById('search-mail').value.toLowerCase();
        const resultsDiv = document.getElementById('search-results');
        if(query.length < 2) { resultsDiv.classList.add('hidden'); return; }

        const filtered = usersDB.filter(u => u.email.toLowerCase().includes(query) && u.email !== currentUser.email);
        resultsDiv.classList.toggle('hidden', filtered.length === 0);
        resultsDiv.innerHTML = filtered.map(u => `
            <div onclick="sendMessageTo('${u.email}')" class="p-4 hover:bg-indigo-50 cursor-pointer flex items-center justify-between border-b border-slate-50 transition-colors">
                <div class="flex items-center gap-3 pr-2 overflow-hidden">
                    <img src="${u.photo}" class="w-8 h-8 rounded-full object-cover border shadow-sm">
                    <span class="text-xs font-bold text-slate-700 truncate">${u.email}</span>
                </div>
                ${u.isVip ? '<span class="badge-vip w-3 h-3"><svg class="w-2.5 h-2.5 text-white" fill="currentColor" viewBox="0 0 20 20"><path d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z"></path></svg></span>' : ''}
            </div>
        `).join('');
    }

    function sendMessageTo(to) {
        const msg = prompt(`Escribe tu mensaje directo para ${to}:`);
        if(!msg) return;
        document.getElementById('email-list').insertAdjacentHTML('afterbegin', `
            <div class="p-5 bg-slate-50 rounded-[2rem] border border-slate-100 text-xs shadow-md slide-in flex gap-4">
                <div class="w-2.5 h-2.5 rounded-full bg-indigo-500 mt-1 flex-shrink-0"></div>
                <div class="flex-1">
                    <b class="text-indigo-500 uppercase block mb-1 font-black">Para: ${to}</b>
                    <p class="text-slate-700 font-medium">${msg}</p>
                </div>
            </div>
        `);
        document.getElementById('search-results').classList.add('hidden');
        document.getElementById('search-mail').value = "";
    }

    function checkAdmin() {
      if(prompt("LLAVE MAESTRA - ACCESO MAESTRO:") === ADMIN_KEY) {
        updateAdminDatabase();
        document.getElementById('admin-panel').classList.remove('hidden');
      } else { alert("Acceso denegado. Llave Maestra incorrecta."); }
    }

    function closeAdmin() { document.getElementById('admin-panel').classList.add('hidden'); }
    function openCompose() { alert("Usa el buscador arriba para enviar un mensaje directo."); }
  </script>
